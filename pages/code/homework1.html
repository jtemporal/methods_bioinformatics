miss.fill <- function(x) {
    if(sum(is.na(as.numeric(x))) ==17 ) {
        x[is.na(x)] <-0
    }
    if(sum(is.na(as.numeric(x))) > 0 & sum(is.na(as.numeric(x))) <17 ) {
        x[is.na(x)] <- mean(as.numeric(x[!is.na(x)]))
    }
    return(x)
}
